const DATA = RAW.map(r => {
  const d = String(r[2]);
  return {
    land: r[0], produkt: r[1], datum: r[2], preis: r[3], menge: r[4],
    umsatz: r[3]*r[4],
    region: DIM_GEO[r[0]]||'Unbekannt',
    kat: DIM_PROD[r[1]]||'Sonstige',
    jahr: d.slice(0,4),
    monLabel: d.slice(0,4)+'-'+d.slice(4,6)
  };
});

let charts = {};
function destroyChart(id) { if(charts[id]){ charts[id].destroy(); delete charts[id]; } }
function groupBy(arr,key) { const m={}; arr.forEach(d=>{m[d[key]]=(m[d[key]]||0)+d.umsatz;}); return m; }
function groupByMenge(arr,key) { const m={}; arr.forEach(d=>{m[d[key]]=(m[d[key]]||0)+d.menge;}); return m; }

function getFiltered() {
  const land = document.getElementById('sl-land').value;
  const kat = document.getElementById('sl-kat').value;
  const jahr = document.getElementById('sl-jahr').value;
  return DATA.filter(d=>(!land||d.land===land)&&(!kat||d.kat===kat)&&(!jahr||d.jahr===jahr));
}

function applyFilters() {
  const fd = getFiltered();
  const parts = [document.getElementById('sl-land').value, document.getElementById('sl-kat').value, document.getElementById('sl-jahr').value].filter(Boolean);
  document.getElementById('filter-badge').textContent = parts.length ? parts.join(' · ') : 'Alle Daten';

  const totalU = fd.reduce((s,d)=>s+d.umsatz,0);
  const totalM = fd.reduce((s,d)=>s+d.menge,0);
  const avgP = fd.length ? Math.round(fd.reduce((s,d)=>s+d.preis,0)/fd.length) : 0;
  document.getElementById('kpi-umsatz').textContent = totalU>=1000000 ? (totalU/1000000).toFixed(2)+' Mio. €' : (totalU/1000).toFixed(0)+' Tsd. €';
  document.getElementById('kpi-trans').textContent = fd.length.toLocaleString('de');
  document.getElementById('kpi-menge').textContent = totalM.toLocaleString('de');
  document.getElementById('kpi-avg').textContent = avgP.toLocaleString('de')+' €';

  // Time chart
  destroyChart('time');
  const byMon={}; fd.forEach(d=>{byMon[d.monLabel]=(byMon[d.monLabel]||0)+d.umsatz;});
  const tLabels=Object.keys(byMon).sort();
  const tVals=tLabels.map(l=>byMon[l]);
  document.getElementById('leg-time').innerHTML='<span><span class="ld" style="background:#0078d4"></span>Monatsumsatz (€)</span>';
  charts['time']=new Chart(document.getElementById('cTime'),{type:'bar',
    data:{labels:tLabels.map(l=>{const p=l.split('-');return p[0].slice(2)+'/'+p[1];}),
      datasets:[{label:'Umsatz',data:tVals,backgroundColor:'#0078d4',borderRadius:3,barPercentage:0.75}]},
    options:{responsive:true,maintainAspectRatio:false,plugins:{legend:{display:false}},
      scales:{x:{ticks:{font:{size:10},maxRotation:45,autoSkip:false}},
              y:{ticks:{callback:v=>v>=1000?(v/1000).toFixed(0)+'K €':v,font:{size:10}}}}}});

  // Region chart
  destroyChart('reg');
  const byReg=groupBy(fd,'region');
  const rLabels=Object.keys(byReg);
  const rVals=rLabels.map(l=>byReg[l]);
  const rTotal=rVals.reduce((s,v)=>s+v,0);
  document.getElementById('leg-reg').innerHTML=rLabels.map((l,i)=>
    `<span><span class="ld" style="background:${COLORS_REG[l]||'#888'}"></span>${l} ${rTotal>0?Math.round(rVals[i]/rTotal*100)+'%':''}</span>`
  ).join('');
  charts['reg']=new Chart(document.getElementById('cReg'),{type:'doughnut',
    data:{labels:rLabels,datasets:[{data:rVals,backgroundColor:rLabels.map(l=>COLORS_REG[l]||'#888'),borderWidth:2,borderColor:'#fff'}]},
    options:{responsive:true,maintainAspectRatio:false,cutout:'55%',plugins:{legend:{display:false}}}});

  // Land chart
  destroyChart('land');
  const byL=groupBy(fd,'land');
  const lE=Object.entries(byL).sort((a,b)=>b[1]-a[1]);
  charts['land']=new Chart(document.getElementById('cLand'),{type:'bar',
    data:{labels:lE.map(e=>e[0]),datasets:[{label:'Umsatz',data:lE.map(e=>e[1]),
      backgroundColor:lE.map(e=>COLORS_LAND[e[0]]||'#888'),borderRadius:3,barPercentage:0.75}]},
    options:{responsive:true,maintainAspectRatio:false,indexAxis:'y',plugins:{legend:{display:false}},
      scales:{x:{ticks:{callback:v=>v>=1000?(v/1000).toFixed(0)+'K':v,font:{size:10}}},
              y:{ticks:{font:{size:10}}}}}});

  // Kat chart
  destroyChart('kat');
  const byK=groupBy(fd,'kat');
  const kLabels=Object.keys(byK);
  const kVals=kLabels.map(l=>byK[l]);
  const kTotal=kVals.reduce((s,v)=>s+v,0);
  charts['kat']=new Chart(document.getElementById('cKat'),{type:'pie',
    data:{labels:kLabels.map((l,i)=>`${l} (${kTotal>0?Math.round(kVals[i]/kTotal*100):'0'}%)`),
      datasets:[{data:kVals,backgroundColor:kLabels.map(l=>COLORS_KAT[l]||'#888'),borderWidth:2,borderColor:'#fff'}]},
    options:{responsive:true,maintainAspectRatio:false,
      plugins:{legend:{position:'bottom',labels:{font:{size:11},padding:10,boxWidth:12}}}}});

  // Prod chart
  destroyChart('prod');
  const byP=groupByMenge(fd,'produkt');
  const pE=Object.entries(byP).sort((a,b)=>b[1]-a[1]).slice(0,8);
  charts['prod']=new Chart(document.getElementById('cProd'),{type:'bar',
    data:{labels:pE.map(e=>e[0].length>14?e[0].slice(0,14)+'…':e[0]),
      datasets:[{label:'Menge',data:pE.map(e=>e[1]),backgroundColor:'#107c41',borderRadius:3,barPercentage:0.75}]},
    options:{responsive:true,maintainAspectRatio:false,indexAxis:'y',plugins:{legend:{display:false}},
      scales:{x:{ticks:{font:{size:10}}},y:{ticks:{font:{size:10}}}}}});
}
